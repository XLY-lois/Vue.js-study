### v-for
##### 循环数组
```
<body>
    <div id="app">
        <p v-for="item in list">{{ item }}</p>
    </div>
</body>
<script>
    var vm = new Vue({
        el:'#app',
        data:{
            list:[1,2,3,4,5,6]
        },
        methods:{}
    });
</script>
```
- 索引值
```
<div id="app">
        <p v-for="(item,i) in list">索引值：{{ i }}---值：{{ item }}</p>
    </div>
```
##### 循环对象数组
```
<body>
    <div id="app">
        <p v-for="user in list1">{{ user.id }} ---- {{ user.name }}</p>
    </div>
</body>
<script>
    var vm = new Vue({
        el:'#app',
        data:{
            list:[1,2,3,4,5,6],
            list1:[
                {id:1,name:'xm1'},
                {id:2,name:'xm2'},
                {id:3,name:'xm3'},
                {id:4,name:'xm4'},
                {id:5,name:'xm5'},
            ]
        },
        methods:{}
    });
</script>
```
##### 循环对象
```
<body>
    <div id="app">
        <p v-for="(val,key) in user">值是：{{ val }} --- 键是 {{ key }}</p>
    </div>
</body>
<script>
    var vm = new Vue({
        el:'#app',
        data:{
            user:{
                id:1,
                name:'tony',
                gender:'男'
            }
        },
        methods:{}
    });
</script>
```
- 除了有val key之外，还有第三个值：索引值

##### 迭代数字
```
<div id="app">
        <p v-for="count in 10">这是第{{ count }}次循环 </p>
    </div>
```
- 从1开始到10
